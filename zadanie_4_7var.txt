String::String(){

};

		for(int i = 0; i < 255; i++)

		S2[i] = '\0';} // Заполнение \0 символом чтобы не было мусора в строках
	cout << "Введите исходную строку:" << endl;

}

void *String::Run(){
	if(strlen(S1) == 10){
		int count = 0;
		for (int i = 0; i < strlen(S1); i++)
			if (S1[i] < 'A' || S1[i] > 'Z')
			{
			S2[count] = S1[i];
			count = count + 1;
			}
	}
	else S2=S1;

	return 0;
}

void *String::Print(){
	cout << "Исходная строка: " << S1 << endl;
	cout << "Получившеяся строка: " << S2 << endl;
	return 0;
}
